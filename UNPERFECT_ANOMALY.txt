            p_pow = 1
            while temp_n % p == 0:

                p_sum += p_pow

            total *= p_sum
        p += 1 if p == 2 else 2
    if temp_n > 1:
        total *= (1 + temp_n)
    return total

def pearson(xs, ys):
    """
    Computes the standard Pearson correlation coefficient between two lists.
    """
    if not xs or not ys or len(xs) != len(ys):
        return 0
    mx = sum(xs) / len(xs)
    my = sum(ys) / len(ys)
    num = sum((x - mx) * (y - my) for x, y in zip(xs, ys))
    denx = math.sqrt(sum((x - mx) ** 2 for x in xs))
    deny = math.sqrt(sum((y - my) ** 2 for y in ys))
    return num / (denx * deny) if denx and deny else 0
